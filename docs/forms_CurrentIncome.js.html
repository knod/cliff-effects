const IncomeForm = function ({ current, time, updateClientValue, snippets }) {

  var type = 'income';

  /** Makes sure values are propagated to 'future' properties if needed */

  var sharedProps = {
    timeState:         current,
    time:              time,
    type:              type,
    updateClientValue: ensureFuture,
  };

  return (

      <IntervalColumnHeadings type={ type } />
